		xsltTemplateProcess		(xsltTransformContextPtr ctxt,

		xsltAttrListTemplateProcess	(xsltTransformContextPtr ctxt,
						 xmlNodePtr target,

}

#endif /* __XML_XSLT_TEMPLATES_H__ */
