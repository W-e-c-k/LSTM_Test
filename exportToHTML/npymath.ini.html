npymath.ini 
[meta]
Name=npymath 
Description=Portable, core math library implementing C99 standard 
Version=0.1 

[variables]
pkgname=numpy.core 
prefix=${pkgdir} 
libdir=${prefix}/lib 
includedir=${prefix}/include 

[default]
Libs=-L${libdir} -lnpymath 
Cflags=-I${includedir} 
Requires=mlib 

[msvc]
Libs=/LIBPATH:${libdir} npymath.lib 
Cflags=/INCLUDE:${includedir} 
Requires=mlib
